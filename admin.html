// server.js const express = require('express'); const cors = require('cors'); const mongoose = require('mongoose'); const multer = require('multer'); const eventRoutes = require('./routes/eventRoutes'); const app = express(); // Middleware app.use(cors()); app.use(express.json()); // Pour analyser le JSON dans le corps des requêtes // Connexion à la base de données mongoose.connect('mongodb://localhost:27017/vinnyworld', { useNewUrlParser: true, useUnifiedTopology: true }).then(() => console.log('Connecté à MongoDB')) .catch(err => console.error('Erreur de connexion à MongoDB:', err)); // Routes app.use('/api/events', eventRoutes); // Démarrage du serveur const PORT = process.env.PORT || 5000; app.listen(PORT, () => { console.log(`Serveur en cours d'exécution sur le port ${PORT}`); });
